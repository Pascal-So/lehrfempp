504  Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic>
505  EvalReferenceShapeFunctions(const Eigen::MatrixXd& refcoords) const override {
506  LF_ASSERT_MSG(refcoords.rows() == 2,
507  "Reference coordinates must be 2-vectors");
508 
509  Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> result(
510  4, refcoords.cols());
511 
512  result.row(0) =
513  ((1 - refcoords.row(0).array()) * (1 - refcoords.row(1).array()))
514  .matrix();
515  result.row(1) =
516  ((refcoords.row(0).array()) * (1 - refcoords.row(1).array())).matrix();
517  result.row(2) =
518  ((refcoords.row(0).array()) * (refcoords.row(1).array())).matrix();
519  result.row(3) =
520  ((1 - refcoords.row(0).array()) * (refcoords.row(1).array())).matrix();
521 
522  return result;
523  }

526  Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic>
527  GradientsReferenceShapeFunctions(
528  const Eigen::MatrixXd& refcoords) const override {
529  LF_ASSERT_MSG(refcoords.rows() == 2,
530  "Reference coordinates must be 2-vectors");
531  size_type n_pts(refcoords.cols());
532 
533  Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> result(4, 2 * n_pts);
534 
535  // reshape the result matrix into a 8xn_pts matrix
536  Eigen::Map<Eigen::Array<SCALAR, Eigen::Dynamic, Eigen::Dynamic>,
537  Eigen::AutoAlign>
538  temp(&result(0, 0), 8, n_pts);
539  temp.row(0) = refcoords.row(1).array() - 1.0;
540  temp.row(1) = 1.0 - refcoords.row(1).array();
541  temp.row(2) = refcoords.row(1).array();
542  temp.row(3) = -refcoords.row(1).array();
543  temp.row(4) = refcoords.row(0).array() - 1.0;
544  temp.row(5) = -refcoords.row(0).array();
545  temp.row(6) = refcoords.row(0).array();
546  temp.row(7) = 1.0 - refcoords.row(0).array();
547 
548  return result;
549  }
