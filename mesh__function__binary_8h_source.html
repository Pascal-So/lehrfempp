94  if constexpr (R1 != Eigen::Dynamic && C1 != Eigen::Dynamic &&
95  R2 != Eigen::Dynamic && C2 != Eigen::Dynamic) { // NOLINT

97  static_assert(R1 == R2, "cannot add matrices with different #rows.");
98  static_assert(C1 == C2, "cannot add matrices with different #cols.");
99 
100  Eigen::Map<const Eigen::Matrix<S1, 1, Eigen::Dynamic>> um(
101  &u[0](0, 0), 1, u.size() * R1 * C1);
102  Eigen::Map<const Eigen::Matrix<S2, 1, Eigen::Dynamic>> vm(
103  &v[0](0, 0), 1, v.size() * R1 * C1);
104 
105  std::vector<Eigen::Matrix<scalar_t, R1, C1>> result(u.size());
106  Eigen::Map<Eigen::Matrix<scalar_t, 1, Eigen::Dynamic>> rm(
107  &result[0](0, 0), 1, u.size() * R1 * C1);
108  rm = um + vm;
109  return result;
110  }
111  if constexpr ((R1 != Eigen::Dynamic && C1 != Eigen::Dynamic) ||
112  (R2 != Eigen::Dynamic && C2 != Eigen::Dynamic)) { // NOLINT
113  // One of the matrices is fixed size:
114  constexpr int fixedRows = std::max(R1, R2);
115  constexpr int fixedCols = std::max(C1, C2);
116 
117  std::vector<Eigen::Matrix<scalar_t, fixedRows, fixedCols>> result(
118  u.size());
119  for (int i = 0; i < u.size(); ++i) {
120  LF_ASSERT_MSG(

123  LF_ASSERT_MSG(
